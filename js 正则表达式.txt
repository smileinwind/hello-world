1.千位符加逗号
	function toThousands(num) {
		return (num || 0).toString().replace(/(\d)(?=(?:\d{3})+$)/g, '$1,');
	}
	小数分隔
	 replace(/(\d)(?=(?:\d{3})+$)/g, '$1,')
	 replace(/\d(?=(?:\d{3})+\b)/g, '$&,')
		11100550.12245 => 11,100,550.12,245
	小数不分隔 
	 replace(/(\d)(?=(?:\d{3})+\.)/g, '$1,')
	 replace(/(\d)(?=(?:\d{3})+\.)/g, '$&,')
		11100550.12245 => 11,100,550.12245
